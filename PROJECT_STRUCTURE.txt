key_algorithms:
  dot_detection:
    type: Pixel pattern analysis
    accuracy: Depends on image quality
    
  symmetry_detection:
    type: Rotational and reflection checking
    angles: [90°, 60°, 45°, 30°]
    scoring: 0-1 normalized
    
  dot_connection:
    type: Nearest neighbor algorithm
    optimization: Distance-based
    constraints: Min/max distance thresholds
    
  complexity_calculation:
    formula: "(dots/50 * 0.6 + lines/100 * 0.4) * 100"

    categories: Simple, Moderate, Complex
